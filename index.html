const FOLDER_ID = ''; // Opcional: ID carpeta por defecto

function doPost(e) {
  try {
    const params = JSON.parse(e.postData.contents);
    const action   = params.action;
    const folderId = params.folderId || FOLDER_ID;

    if (action === 'save') {
      const folder   = DriveApp.getFolderById(folderId);
      const fileName = params.fileName;
      const content  = params.content;
      const iter = folder.getFilesByName(fileName);
      if (iter.hasNext()) {
        iter.next().setContent(content);
      } else {
        folder.createFile(fileName, content, MimeType.PLAIN_TEXT);
      }
      return out({ ok: true, fileName });

    } else if (action === 'list') {
      const folder = DriveApp.getFolderById(folderId);
      const iter   = folder.getFilesByType(MimeType.PLAIN_TEXT);
      const files  = [];
      while (iter.hasNext()) {
        const f = iter.next();
        files.push({ id: f.getId(), name: f.getName(),
                     modifiedTime: f.getLastUpdated().toISOString(),
                     size: f.getSize() });
      }
      files.sort((a,b) => b.modifiedTime.localeCompare(a.modifiedTime));
      return out({ ok: true, files });

    } else if (action === 'read') {
      const file = DriveApp.getFileById(params.fileId);
      return out({ ok: true, content: file.getBlob().getDataAsString() });

    } else if (action === 'getImage') {
      // Proxy de imágenes: evita CORS al descargar imágenes de Drive para el PDF
      const file     = DriveApp.getFileById(params.fileId);
      const blob     = file.getBlob();
      const base64   = Utilities.base64Encode(blob.getBytes());
      const mimeType = blob.getContentType();
      return out({ ok: true, base64: base64, mimeType: mimeType });
    }

    return out({ ok: false, error: 'Acción desconocida' });
  } catch(err) {
    return out({ ok: false, error: err.message });
  }
}

function doGet(e) {
  return out({ ok: true, msg: 'App Script activo' });
}

function out(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

⚠ Limitación CORS (archivo local)

Al abrir este archivo con file://, el navegador bloquea todos los fetch externos. La solución es un Worker proxy propio en Cloudflare que añade cabeceras CORS.

Si se configura, todas las llamadas de IA pasarán por este Worker. Deja vacío para llamada directa.

Código del Worker (copia y despliega)
  1. Ve a Dashboard → Workers → Crear
  2. Pega el código, haz clic en Deploy
  3. En Settings → Variables agrega ACCOUNT_ID y API_TOKEN como secretos
  4. Copia la URL y pégala arriba
export default {
  async fetch(req, env) {
    const cors = {
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'POST, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type',
    };
    if (req.method === 'OPTIONS')
      return new Response(null, { headers: cors });
    const { model, messages } = await req.json();
    const url = `https://api.cloudflare.com/client/v4/accounts/`
              + `${env.ACCOUNT_ID}/ai/run/${model}`;
    const r = await fetch(url, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${env.API_TOKEN}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({ messages }),
    });
    return Response.json(await r.json(), { headers: cors });
  },
};
